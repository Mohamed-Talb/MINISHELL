#include "../minishell.h"

void ft_putnstr_fd(char *str, int n, int fd)
{
	int i;

	i = 0;
	while (i < n)
	{
		write(fd, &str[i], 1);
		i++;
	}
}

void checkerrors(t_data *data, char c)
{
	if (c == '\0' || in_set(REDIRECTION_SET , c) == 1)
	{
		ft_putstr_fd(UNEXPECTED_TOKEN, 2);
		if (c == '\0')
			ft_putstr_fd("newline", 2);
		else
			ft_putchar_fd(c, 2);
		errors(data, "'\n", 1);
	}
}

int redirect_helper(t_data *data, t_dlist *token, char *s, int i)
{
	while (ft_iswhitespace(s[i]))
		i++;
	checkerrors(data, s[i]);
	while (s[i] != '\0' && in_set(REDIRECTION_SET, s[i]) != 1 && !ft_iswhitespace(s[i]))
	{
		if(s[i] == '"' && (token->type == RIGHT_RED
			|| token->type == LEFT_RED || token->type == RIGHT_HER))
				i = double_q(data, token, s, i);
		if (s[i] == '$' && token->type != LEFT_HER)
			i = expand(data, token->content, s, i);
		else if(s[i] == '\'')
			i = single_q(data, token->content, s, i);
		else
		{
			token->content = ft_append(token->content, s[i], -1);
			i++;
		}
	}
	printf("delemiter: %s\n", (char *) token->content);
	return (i);
}

int redirect(t_data *data, t_dlist *token, char *s, int i)
{
	// if(ft_strlen(token->content) != 0)
	// 	ft_dlstback(&data->cmd_list, ft_strdup(""));
	// token = ft_dlstlast(data->cmd_list); // why is this necessary, you're not even setting the type?
	if (s[i] == '>')
	{
		token->type = RIGHT_RED;
		i++;
		if (s[i] == '>')
		{
			token->type = RIGHT_HER;
			i++;
		}
	}
	else if (s[i] == '<')
	{
		token->type = LEFT_RED;
		i++;
		if (s[i] == '<')
		{
			token->type = LEFT_HER;
			i++;
		}
	}
	return (redirect_helper(data, token, s, i));
}
